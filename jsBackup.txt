var exec = require('cordova/exec');

function CoolPlugin() { 
 console.log("CoolPlugin.js: is created");
}

CoolPlugin.prototype.regUser = function(caller){

exec(function(result){


     alert("Register successfull");

     if(result == "RING_RING"){

        alert("RINGING");

     }

   },
  function(result){


    alert("Error register");
  },"CoolPlugin","REGISTER",[caller]);

};

CoolPlugin.prototype.callUser = function(callee){

    exec(function(result){
        alert("Call successfull");
    },
    function(result){
        alert("Error call user");
    },"CoolPlugin","CALL",[callee]);
};
